Introduction

Oul is a statically typed, interpreted language written in Rust.

  • Built-in 2D and 3D graphics
  • Custom shaders
  • OBJ model loading
  • No semicolons
  • Strict types

Calculator

import olstream

fn main() {
    int a = to_int(input("a: "))
    string op = input("op: ")
    int b = to_int(input("b: "))

    if (op == "+") { olstream::println(a + b) }
    else if (op == "-") { olstream::println(a - b) }
    else if (op == "*") { olstream::println(a * b) }
    else if (op == "/") { olstream::println(a / b) }
}

2D Game

import olgfx

fn main() {
    olgfx::open(800, 600, "Game")
    int x = 400
    int y = 300

    while (olgfx::is_open()) {
        olgfx::clear(0, 0, 0)
        if (olgfx::key_down("W")) { y = y - 3 }
        if (olgfx::key_down("S")) { y = y + 3 }
        if (olgfx::key_down("A")) { x = x - 3 }
        if (olgfx::key_down("D")) { x = x + 3 }
        olgfx::fill_circle(x, y, 20, 0, 200, 255)
        olgfx::update()
        olgfx::sleep(16)
    }
    olgfx::close()
}

3D Demo

import olgfx
import ol3d

fn main() {
    olgfx::open(800, 600, "3D")
    ol3d::screen_size(800, 600)
    ol3d::shader("phong")

    int cube = ol3d::create_cube()
    ol3d::mesh_color(cube, 0.2, 0.6, 1.0)
    float yaw = 0.0

    while (olgfx::is_open()) {
        ol3d::tick()
        olgfx::clear(15, 15, 25)
        yaw = yaw + ol3d::delta_time()
        ol3d::camera_pos(0.0, 0.0, -5.0)
        ol3d::camera_rot(yaw, 0.3)
        ol3d::render(cube)
        olgfx::update()
        olgfx::sleep(1)
    }
    olgfx::close()
}
