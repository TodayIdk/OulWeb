import olgfx
import ol3d

fn main() {
    olgfx::open(800, 600, "My Game")
    int cube = ol3d::create_cube()
    ol3d::shader("phong")

    while (olgfx::is_open()) {
        olgfx::clear(0, 0, 0)
        ol3d::render(cube)
        olgfx::update()
    }
}
